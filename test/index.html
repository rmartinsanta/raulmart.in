function get(url, callback)
{
    let xmlhttp=new XMLHttpRequest();
    xmlhttp.onreadystatechange=function(){
        if (xmlhttp.readyState == XMLHttpRequest.DONE)
        {
            callback(xmlhttp);
        }
    }
    xmlhttp.open("GET", url);
    xmlhttp.send();
